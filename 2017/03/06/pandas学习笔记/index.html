test = Series([1,2,3,4], ['a', 'b', 'c', 'd'])

a    1
b    2
c    3
d    4
dtype: int64

test.values
array([1, 2, 3, 4])

test.index
Index([u'a', u'b', u'c', u'd'], dtype='object')

test[test > 2]
Out[18]:
c    3
d    4
dtype: int64

DataFrame

既有行索引，也有列索引，可以看作是由Series组成的字典

# 简单创建
In [19]: data = {'id': [0,1,2], 'name': ['jack', 'bob', 'ali'], 'age': [12,32,11]}

In [20]: data
Out[20]: {'age': [12, 32, 11], 'id': [0, 1, 2], 'name': ['jack', 'bob', 'ali']}

In [21]: test = DataFrame(data)

In [22]: test
Out[22]:
age  id  name
0   12   0  jack
1   32   1   bob
2   11   2   ali

# 也可以指定列索引，按照顺序创建
In [23]: test = DataFrame(data, columns = ['age', 'name', 'id'])

In [24]: test
Out[24]:
age  name  id
0   12  jack   0
1   32   bob   1
2   11   ali   2

# 不存在的列，例如wahaha，会产生NA值，创建时行索引可以用index指定
In [25]: test = DataFrame(data, columns = ['age', 'name', 'id', 'wahaha'], index = ['one', 'two', 'three'])

In [26]: test
Out[26]:
age  name  id wahaha
one     12  jack   0    NaN
two     32   bob   1    NaN
three   11   ali   2    NaN

# 行列索引
In [31]: test.index
Out[31]: Index([u'one', u'two', u'three'], dtype='object')
In [32]: test.columns
Out[32]: Index([u'age', u'name', u'id', u'wahaha'], dtype='object')

# values，以二维数组形式返回
In [41]: test.values
Out[41]:
array([[12, 'jack', 0],
       [32, 'bob', 1],
       [11, 'ali', 2]], dtype=object)

# 列引用，输出是普通的Series对象
In [33]: test.age
Out[33]:
one      12
two      32
three    11
Name: age, dtype: int64

# 行引用，输出是普通的Series对象
In [34]: test.ix['two']
Out[34]:
age        32
name      bob
id          1
wahaha    NaN
Name: two, dtype: object

# 通过赋值批量设置列值、添加新列
In [35]: test['wahaha'] = 3
In [36]: test
Out[36]:
age  name  id  wahaha
one     12  jack   0       3
two     32   bob   1       3
three   11   ali   2       3

# 删除整列
del test['wahaha']

# 转置
In [38]: test
Out[38]:
       age  name  id
one     12  jack   0
two     32   bob   1
three   11   ali   2
In [39]: test.T
Out[39]:
one  two three
age     12   32    11
name  jack  bob   ali
id       0    1     2

# 按照逗号进行分割，默认以第一行为列名
df = pd.read_csv('file')

# read_table更灵活，可以指定分隔符，此处的分隔符是正则表达式，匹配任意多个空格
df = pd.read_table('file', sep = '/s+')

# 参数skiprows = [0, 2, 3]，可以指定跳过的行数
# 参数na_values = ['NULL']，可以指定缺失值有哪些，pandas使用nan来占位
# 参数nrows = 5，可以指定只读取一定量的行

# 读取excel文件

数据合并

In [3]: df1 = DataFrame({'key': ['b', 'b', 'a', 'c', 'a', 'b'], 'data1': range(6)})

In [4]: df1
Out[4]: 
   data1 key
0      0   b
1      1   b
2      2   a
3      3   c
4      4   a
5      5   b

In [5]: df2 = DataFrame({'key': ['a', 'b', 'a', 'b', 'd'], 'data2': range(5)})

In [6]: df2
Out[6]: 
   data2 key
0      0   a
1      1   b
2      2   a
3      3   b
4      4   d

# 默认是笛卡尔乘积，on可以传入数组，以多个键连接
# 如果左右列名不一样，可以用left_on、right_on
# sort属性默认为True，处理大数据时可以设置为False，增强性能
In [8]: pd.merge(df1, df2, on = 'key')
Out[8]: 
   data1 key  data2
